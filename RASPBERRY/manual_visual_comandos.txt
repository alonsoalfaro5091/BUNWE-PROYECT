import pymysql
from fastapi import FastAPI, HTTPException

app = FastAPI()

def conectar_bd():
    return pymysql.connect(
        host='localhost', user='faro', password='TU_PASSWORD', db='sleep_db', autocommit=True
    )

# 1. RUTA GET (CONSULTAR)
@app.get("/historial")
def obtener_historial():
    db = conectar_bd()
    cursor = db.cursor(pymysql.cursors.DictCursor)
    cursor.execute("SELECT id, name, horas FROM registro_sueno")
    datos = cursor.fetchall()
    db.close()
    return {"historial": datos}

# 2. RUTA POST (GUARDAR)
@app.post("/guardar")
def guardar_registro(nombre: str, horas: float, fecha: str):
    if horas < 0 or horas > 24:
        raise HTTPException(status_code=400, detail="Horas inválidas (0 a 24)")
    db = conectar_bd()
    cursor = db.cursor()
    cursor.execute("INSERT INTO registro_sueno (name, horas, fecha) VALUES (%s, %s, %s)", (nombre, horas, fecha))
    db.close()
    return {"status": "success"}
